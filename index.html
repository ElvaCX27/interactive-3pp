        <p>Player 1's Points: <span id="player1-points">20</span></p>

        <p>Player 2's Points: <span id="player2-points">60</span></p>

    </div>


    <script>

        let player1Points = 20;

        let player2Points = 60;

        let totalPoints = 40;


function allocatePoints() {

    if (totalPoints > 0) {

        console.log("Coin clicked! Total points left: " + totalPoints);  // Check if click is detected


        // Randomly allocate points to Player 1 or Player 2

        let allocation = Math.random() > 0.5 ? 'player1' : 'player2';

        if (allocation === 'player1') {

            player1Points += 10;

            document.getElementById('player1-points').textContent = player1Points;

            console.log("Points allocated to Player 1: " + player1Points);  // Log player 1's updated points

        } else {

            player2Points += 10;

            document.getElementById('player2-points').textContent = player2Points;

            console.log("Points allocated to Player 2: " + player2Points);  // Log player 2's updated points

        }

        totalPoints -= 10;

        

        // Remove the clicked coin

        document.querySelector('.coin').remove();

    }

}


    </script>
